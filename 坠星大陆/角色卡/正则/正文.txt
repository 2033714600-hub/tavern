    <script>
        /* ========== 主题切换功能 ========== */
        function toggleTheme() {
            var container = document.getElementById('mainContainer');
            var btnIcon = document.getElementById('themeIcon');
            var btnText = document.getElementById('themeBtn');

            container.classList.toggle('dark-mode');

            if (container.classList.contains('dark-mode')) {
                btnText.innerHTML = '<span id="themeIcon">☀️</span> 开灯';
            } else {
                btnText.innerHTML = '<span id="themeIcon">🌙</span> 关灯';
            }
        }

        /* ========== 获取消息内容 ========== */
        function getMessageData() {
            var chatMessages = getChatMessages(getCurrentMessageId());
            if (!chatMessages || chatMessages.length === 0) {
                return null;
            }
            return chatMessages[0].message;
        }

        /* ========== 提取外层正文 ========== */
        function extractContent(messageText) {
            var match = messageText.match(/(?:<|&lt;)story(?:>|&gt;)([\s\S]*?)(?:<|&lt;)\/story(?:>|&gt;)/);
            if (match && match[1]) {
                return match[1].trim();
            }
            return messageText;
        }

        /* ========== 渲染普通正文段落 ========== */
        function renderNormalText(text) {
            if (!text || !text.trim()) return '';
            var paragraphs = text.split(/\n\s*\n/);
            var html = '';

            paragraphs.forEach(function(p) {
                var lines = p.split('\n');
                lines.forEach(function(line) {
                    /* 去除HTML标签和多余空格 */
                    var cleanLine = line.replace(/<br\s*\/?>/gi, '').trim();
                    if (cleanLine) {
                        /* 识别对话并渲染 */
                        if (cleanLine.startsWith('“') || cleanLine.startsWith('”') || cleanLine.startsWith('"') || cleanLine.startsWith('「')) {
                            html += '<p class="dialogue">' + cleanLine + '</p>';
                        } else {
                            html += '<p class="narrative">' + cleanLine + '</p>';
                        }
                    }
                });
            });
            return html;
        }

        /* ========== 渲染NPC卡片 ========== */
        function renderNpcPanel(content) {
            var result = {};
            var values = [];

            /* 提取所有方括号内的内容 */
            var regex = /\[(.*?)\]/g;
            var match;
            while ((match = regex.exec(content)) !== null) {
                values.push(match[1].trim());
            }

            /* 按设定的固定顺序赋值 */
            if (values.length >= 6) {
                result['姓名'] = values[0];
                result['种族'] = values[1];
                result['饱食度'] = values[2].replace('|', ' - ');
                result['忠诚度'] = values[3].replace('|', ' - ');
                result['技能'] = values[4];
                result['行为'] = values[5];
            } else {
                /* 兜底，万一格式不对 */
                result['姓名'] = values[0] || '未知名字';
                result['种族'] = values[1] || '未知种族';
                result['饱食度'] = values[2] ? values[2].replace('|', ' - ') : '???';
            }

            /* 拼接卡片 HTML */
            var html = '<div class="npc-panel-container">' +
                '<div class="npc-panel-header">' +
                '<span class="npc-panel-name">🐾 【' + (result['姓名'] || '未知名字') + '】</span>' +
                '<span class="npc-panel-race">' + (result['种族'] || '未知种族') + '</span>' +
                '</div>' +
                '<div class="npc-panel-stats">' +
                '<div class="npc-panel-stat-row">' +
                '<span class="npc-panel-label">饱食度</span>' +
                '<span class="npc-panel-value-hunger">' + (result['饱食度'] || '???') + '</span>' +
                '</div>' +
                '<div class="npc-panel-stat-row">' +
                '<span class="npc-panel-label">忠诚度</span>' +
                '<span class="npc-panel-value-loyal">' + (result['忠诚度'] || '???') + '</span>' +
                '</div>' +
                '</div>' +
                '<div class="npc-panel-skills-box">' +
                '<div class="npc-panel-skills-title">🌿 生存技能</div>' +
                '<div class="npc-panel-skills-text">' + (result['技能'] || '暂无数据') + '</div>' +
                '</div>' +
                '<div class="npc-panel-behavior">' +
                '<strong>初见行为：</strong>' + (result['行为'] || '没有记录哦') +
                '</div>' +
                '</div>';

            return html;
        }

        /* ========== 综合解析与渲染 ========== */
        function parseAndRenderAll(text) {
            var html = '';

            /* 使用正则寻找嵌在正文里的 npc_panel */
            var npcRegex = /(?:<|&lt;)npc_panel(?:>|&gt;)([\s\S]*?)(?:<|&lt;)\/npc_panel(?:>|&gt;)/g;
            var lastIndex = 0;
            var match;

            while ((match = npcRegex.exec(text)) !== null) {
                /* 1. 渲染卡片前面的普通正文 */
                var normalText = text.substring(lastIndex, match.index);
                html += renderNormalText(normalText);

                /* 2. 渲染 NPC 卡片 */
                var panelContent = match[1];
                html += renderNpcPanel(panelContent);

                lastIndex = npcRegex.lastIndex;
            }

            /* 3. 渲染最后剩下的普通正文 */
            if (lastIndex < text.length) {
                var remainingText = text.substring(lastIndex);
                html += renderNormalText(remainingText);
            }

            document.getElementById('content').innerHTML = html || '<p class="narrative">空空的草地，什么都没有呢。</p>';
        }

        /* ========== 主函数 ========== */
        function init() {
            try {
                var messageText = getMessageData();
                if (!messageText) {
                    document.getElementById('content').innerHTML =
                        '<div class="error-msg">❌ 找不到消息内容啦。</div>';
                    return;
                }

                /* 提取最外层的 story 标签内容 */
                var text = extractContent(messageText);

                /* 综合渲染 */
                parseAndRenderAll(text);

            } catch (error) {
                document.getElementById('content').innerHTML =
                    '<div class="error-msg">❌ 渲染失败：' + error.message + '</div>';
            }
        }

        $(function() { init(); });
    </script>
</body>
</html>
